CSS reset就像是一种宁可错杀三千不可放过一个的做法。
 *{ margin:0; padding:0; }

总结一下

 ul, ol, fieldset{ margin: 0; padding: 0; }
 th, td, legend{ padding:0; }
 body,h1, h2, h3, h4, h5, h6, p, dl,dd,form,textarea { margin:0; }

 /*隐藏横向滚动，垂直滚动根据内容自适应（去除IE默认垂直滚动条）*/   
 html { overflow-x: hidden; overflow-y: auto; }
 /* 重置列表元素 */
 li { list-style: none; }
 /*去掉下滑线*/  
 a { text-decoration: none; }
 /*:link, :visited { text-decoration: none; } 真正有下划线样式的其实只有 :link和:visited*/
 a, button{ cursor:pointer; }
 /* 去掉边框 IE8-要在ifarme标签上添加属性 frameborder="0" 才能去掉边框 */
 fieldset, img ,iframe { border: 0; } 

 /*去除字体默认样式*/
 i,em,dfn,var,cite,address,optgroup{ font-style:normal; }
 strong,th{ font-weight:normal; }
 small,button, input, select, textarea{ font-size:100%; }
 //code,pre,samp,kbd{ font-size:100%; }
 hr{ margin:0; border:1px solid; border-width:1px 0 0 ; height:0; clear:both;  }

 /*继承父级元素的字体大小，粗体还原*/
 h1, h2, h3, h4, h5, h6{ font-size:100%; font-weight:normal; } 
 /* 让表格边框可以为1 */
 table { border-collapse: collapse; border-spacing: 0; } 
 caption, th { text-align: left; } 

 button, input, select { vertical-align: middle; }
 textarea{ vertical-align: top; }
